#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n, size;
	cin >> n >> size;
	if (n == 0) {
		cout << "Hash Table is empty";
		return 0;
	}
	vector<int> table(size, -1);
	vector<int> keys(n);
	for (int i = 0; i < n; i++) {
		cin >> keys[i];
	}
	for (int k = 0; k < n; k++) {
		int key = keys[k];
		int index = key % size;
		bool placed = false;


		for (int i = 0; i < size; i++) {
		int newIndex = (index + i * i) % size;
			if (table[newIndex] == -1) {
			table[newIndex] = key;
				placed = true;
		break;
		}
		}

		if (!placed) {
		cout << "Hash table is full. Cannot insert records from key " << key<<"\n";
			//return 0;
			break;
		}
		
	}


	for (int i = 0; i < size; i++) {
		cout << "T[" << i << "] -> ";
		if (table[i] == -1) cout << -1;
		else cout << table[i];
		cout << endl;
	}

	return 0;
}
